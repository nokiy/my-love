    <script>

        const answerInput = document.getElementById('answerInput');

        const revealButton = document.getElementById('revealButton');

        const confettiCanvas = document.getElementById('confettiCanvas');

        const ctx = confettiCanvas.getContext('2d');


        function resizeCanvas() {

            confettiCanvas.width = confettiCanvas.offsetWidth;

            confettiCanvas.height = confettiCanvas.offsetHeight;

        }

        resizeCanvas();

        window.addEventListener('resize', resizeCanvas);


        const colors = ['#ff0000', '#00ff00', '#0000ff', '#ffff00', '#ff00ff', '#00ffff', '#ffa500', '#8a2be2'];


        class Confetti {

            constructor(x, y) {

                this.x = x;

                this.y = y;

                this.color = colors[Math.floor(Math.random() * colors.length)];

                this.size = Math.random() * 10 + 5;

                this.speedY = -Math.random() * 3 - 3;

                this.speedX = Math.random() * 6 - 3;

                this.angle = Math.random() * 360;

                this.spin = Math.random() * 0.2 - 0.1;

                this.shape = Math.random() < 0.5 ? 'circle' : 'rectangle';

                this.opacity = 1;

            }


            update() {

                this.y += this.speedY;

                this.x += this.speedX;

                this.angle += this.spin;

                this.speedY += 0.1;

                this.opacity -= 0.02;

                this.opacity = Math.max(0, this.opacity);

            }


            draw() {

                ctx.save();

                ctx.translate(this.x, this.y);

                ctx.rotate(this.angle * Math.PI / 180);

                ctx.fillStyle = this.color;

                ctx.globalAlpha = this.opacity;

                if (this.shape === 'circle') {

                    ctx.beginPath();

                    ctx.arc(0, 0, this.size / 2, 0, Math.PI * 2);

                    ctx.fill();

                } else {

                    ctx.fillRect(-this.size / 2, -this.size / 2, this.size, this.size);

                }

                ctx.restore();

            }

        }


        let confetti = [];


        function createConfetti() {

            for (let i = 0; i < 200; i++) {

                confetti.push(new Confetti(Math.random() * confettiCanvas.width, confettiCanvas.height));

            }

        }


        function animateConfetti() {

            ctx.clearRect(0, 0, confettiCanvas.width, confettiCanvas.height);

            confetti.forEach((c, index) => {

                c.update();

                c.draw();

                if (c.opacity <= 0 || c.y > confettiCanvas.height) {

                    confetti.splice(index, 1);

                }

            });

            if (confetti.length > 0) {

                requestAnimationFrame(animateConfetti);

            }

        }


        revealButton.addEventListener('click', () => {

            answerInput.value = '是你呀，bb!';

            createConfetti();

            animateConfetti();

            revealButton.style.transform = 'scale(1.1)';

            setTimeout(() => {

                revealButton.style.transform = 'scale(1)';

            }, 200);

        });
